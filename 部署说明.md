# COREONE 部署指南

> 适用于公司 Linux 服务器的 Docker 部署

## 服务器要求

- **OS**: Linux (Ubuntu 22.04+ / CentOS 8+ / Debian 12+)
- **Docker**: 20.10+
- **Docker Compose**: 2.0+
- **端口**: 1 个可用端口（默认 8080，可自定义）
- **内存**: 最低 512MB，建议 1GB+
- **磁盘**: 1GB+（含 SQLite 数据库增长空间）

## 快速部署（3 步）

### 1. 上传代码到服务器

```bash
set -euo pipefail
# 在本地打包项目（Windows 用 PowerShell / Git Bash；必须在 git 仓库根执行）
cd d:/Git/COREONE/最新代码
# 必须填入 PM 已批准且 CI 全绿的完整 40 位提交 SHA；不允许用漂移的 HEAD/分支名。
APPROVED_SHA='<approved-40-character-commit-sha>'
test "$(git rev-parse "${APPROVED_SHA}^{commit}")" = "$APPROVED_SHA" || exit 1
test -z "$(git status --porcelain)" || { echo '❌ 工作区不干净，停止打包' >&2; exit 1; }

# 只从已批准提交的已跟踪树打包；不要用 --worktree-attributes，否则未提交的属性可改变包内容。
DEPLOY_ARCHIVE="coreone-deploy-${APPROVED_SHA}.tar.gz"
git archive -o "$DEPLOY_ARCHIVE" "$APPROVED_SHA" \
  前端代码 后端代码 docker-compose.yml 部署说明.md

# fail-closed 检查包内不含秘密、数据库、日志或危险一次性脚本；只允许精确的
# .env.example/.env.sample/.env.template 文档模板（其内容已由批准 SHA 的 secret-scan 审过）。
if tar tzf "$DEPLOY_ARCHIVE" \
  | grep -Ev '(^|/)\.env\.(example|sample|template)$' \
  | grep -Eq '(^|/)(\.env($|\.)|token[^/]*\.json$|[^/]*\.(db|sqlite|sqlite3|pem|key|p12|pfx|jks|keystore|log)$|fix-[^/]*$|test-sqlite\.cjs$)'; then
  echo '❌ 部署包含敏感或危险工件，勿上传！' >&2
  exit 1
fi
sha256sum "$DEPLOY_ARCHIVE" | tee "${DEPLOY_ARCHIVE}.sha256"

# 上传到服务器（替换为你的服务器地址）
scp "$DEPLOY_ARCHIVE" "${DEPLOY_ARCHIVE}.sha256" root@your-server-ip:/opt/
```

### 2. 服务器端解压并启动

先单独执行 `ssh root@your-server-ip` 并确认已经进入远端 shell；随后只在该远端 shell 中粘贴下面代码。
不要把 SSH 命令和远端命令作为一个代码块整体执行，否则 SSH 退出后余下命令会落回本机。

```bash
set -euo pipefail
# 校验并解压到不可变 release 目录；APPROVED_SHA 必须与本地审批记录一致。
cd /opt
APPROVED_SHA='<approved-40-character-commit-sha>'
DEPLOY_ARCHIVE="coreone-deploy-${APPROVED_SHA}.tar.gz"
sha256sum -c "${DEPLOY_ARCHIVE}.sha256"
RELEASE_DIR="/opt/coreone/releases/${APPROVED_SHA}"
test ! -e "$RELEASE_DIR" || { echo '❌ 该 release 已存在，停止覆盖' >&2; exit 1; }
mkdir -p "$RELEASE_DIR"
tar -xzvf "$DEPLOY_ARCHIVE" -C "$RELEASE_DIR"
# 在全部验收通过前不切换 current 指针；失败时审批记录仍指向上一个已知 release。
cd "$RELEASE_DIR"
export COMPOSE_PROJECT_NAME=coreone
export COREONE_IMAGE_TAG="$APPROVED_SHA"
printf '%s\n' "$APPROVED_SHA" > DEPLOYED_SHA

# ⚠️ 启动前从正式密钥管理器读取已批准的强随机 JWT secret；本流程不生成、展示或落盘。
read -rsp "JWT secret from the approved secret manager: " JWT_SECRET && echo
test -n "$JWT_SECRET" || { echo '❌ 未读取到 JWT secret' >&2; exit 1; }
export JWT_SECRET
export COREONE_ALLOW_DATABASE_CREATE=1

# 明确记录本次批准的宿主发布端口，避免依赖未纳入 release 的旧 .env 或隐式默认值。
read -rp "Approved frontend host port [8080]: " COREONE_PORT
COREONE_PORT="${COREONE_PORT:-8080}"
case "$COREONE_PORT" in ''|*[!0-9]*) echo '❌ frontend 端口必须是数字' >&2; exit 1 ;; esac
[ "$COREONE_PORT" -ge 1 ] && [ "$COREONE_PORT" -le 65535 ] \
  || { echo '❌ frontend 端口必须在 1..65535' >&2; exit 1; }
export COREONE_PORT

# “快速部署”只适用于全新主机。已存在容器或生产卷时必须走下文旧库升级，不能误建/覆盖数据库。
EXISTING_COREONE_CONTAINERS="$(docker ps -aq --filter "label=com.docker.compose.project=${COMPOSE_PROJECT_NAME}")"
if [ -n "$EXISTING_COREONE_CONTAINERS" ] \
  || docker container inspect coreone-backend >/dev/null 2>&1 \
  || docker container inspect coreone-frontend >/dev/null 2>&1 \
  || docker volume inspect "${COMPOSE_PROJECT_NAME}_coreone-data" >/dev/null 2>&1; then
  echo '❌ 检测到既有 COREONE 容器或数据卷；请改走“旧生产库安全升级”。' >&2
  exit 1
fi
unset EXISTING_COREONE_CONTAINERS

# 全新生产库必须受控创建初始管理员，以便在切流前验证关键登录路径。
# 输入不回显，实际口令不会写进 shell 历史；须为 12 个以上 Unicode 字符、UTF-8 不超过 72 字节，
# 且通过字符多样性、重复/顺序/常见弱口令与已泄露值检查。
read -rsp "Initial admin password: " ADMIN_INITIAL_PASSWORD && echo
test -n "$ADMIN_INITIAL_PASSWORD" || { echo '❌ 全新生产部署必须设置初始管理员口令' >&2; exit 1; }
export ADMIN_INITIAL_PASSWORD

# 从开始创建资源起，任一步失败都移除候选容器并保持服务离线；绝不删除 volume，也不自动启动旧镜像。
DEPLOYMENT_COMPLETE=0
deployment_abort() {
  local status=$?
  local cleanup_ok=1
  trap - EXIT HUP INT TERM
  if [ "${DEPLOYMENT_COMPLETE:-0}" -ne 1 ]; then
    echo '❌ 新装验收未完成；正在停止候选容器，数据卷保留供排查。' >&2
    if ! docker compose down --remove-orphans; then
      echo '⚠️ compose down 失败，尝试按固定容器名强制停止。' >&2
      docker rm -f coreone-frontend coreone-backend >/dev/null 2>&1 || cleanup_ok=0
    fi
    if ! docker info >/dev/null 2>&1; then
      cleanup_ok=0
    else
      for container in coreone-frontend coreone-backend; do
        if docker container inspect "$container" >/dev/null 2>&1 \
          && [ "$(docker inspect --format '{{.State.Running}}' "$container")" = 'true' ]; then
          cleanup_ok=0
        fi
      done
    fi
    if [ "$cleanup_ok" -ne 1 ]; then
      echo '🚨 无法确认候选容器已停止；保持外部流量隔离并人工处置，不能宣称回滚完成。' >&2
    fi
  fi
  unset ADMIN_INITIAL_PASSWORD ADMIN_VERIFY_PASSWORD JWT_SECRET COREONE_ALLOW_DATABASE_CREATE
  [ "$status" -ne 0 ] || status=1
  exit "$status"
}
trap deployment_abort EXIT HUP INT TERM

# 兼容 Compose 2.0+的 backend 健康等待（最长 120 秒）。
wait_backend_healthy() {
  local status
  for _ in $(seq 1 60); do
    status="$(docker inspect --format '{{if .State.Health}}{{.State.Health.Status}}{{else}}{{.State.Status}}{{end}}' \
      coreone-backend 2>/dev/null || true)"
    [ "$status" = "healthy" ] && return 0
    case "$status" in exited|dead) return 1 ;; esac
    sleep 2
  done
  return 1
}

# 先构建并固定本次 SHA 的两个镜像；frontend 在 backend 安全验收前不启动。
# ⚠️ 若复用旧数据库 volume，先不要执行这一段；须先完成下文“旧生产库安全升级”。
docker compose build backend frontend
BACKEND_IMAGE_ID="$(docker image inspect --format '{{.Id}}' "coreone-backend:${APPROVED_SHA}")"
FRONTEND_IMAGE_ID="$(docker image inspect --format '{{.Id}}' "coreone-frontend:${APPROVED_SHA}")"
test -n "$BACKEND_IMAGE_ID" && test -n "$FRONTEND_IMAGE_ID"

ADMIN_VERIFY_PASSWORD="$ADMIN_INITIAL_PASSWORD"
docker compose up -d --no-deps --no-build backend
wait_backend_healthy
test "$(docker inspect --format '{{.Image}}' coreone-backend)" = "$BACKEND_IMAGE_ID"

# 首次建库成功后，无论是否创建 admin，都立即移除两个一次性开关并强制重建 backend。
# 第二个容器必须在“数据库已存在 + 禁止建空库 + 不携带初始口令”条件下再次健康。
unset ADMIN_INITIAL_PASSWORD COREONE_ALLOW_DATABASE_CREATE
ADMIN_INITIAL_PASSWORD='' COREONE_ALLOW_DATABASE_CREATE=0 \
  docker compose up -d --no-deps --no-build --force-recreate backend
wait_backend_healthy
test "$(docker inspect --format '{{.Image}}' coreone-backend)" = "$BACKEND_IMAGE_ID"

# 同一批准镜像在生产模式下故意不给 JWT_SECRET，必须因 JWT 门禁及时失败，不能因缺库或超时假绿。
NO_SECRET_LOG="$(mktemp)"
set +e
timeout 30s docker run --rm --network none \
  -e NODE_ENV=production -e JWT_SECRET= -e DATABASE_PATH=/tmp/coreone-no-secret.db \
  -e COREONE_ALLOW_DATABASE_CREATE=1 \
  "$BACKEND_IMAGE_ID" >"$NO_SECRET_LOG" 2>&1
NO_SECRET_STATUS=$?
set -e
if [ "$NO_SECRET_STATUS" -eq 0 ] || [ "$NO_SECRET_STATUS" -eq 124 ] \
  || ! grep -Eq 'JWT_SECRET environment variable is required|JWT_SECRET.*(缺失|必须)' "$NO_SECRET_LOG"; then
  rm -f "$NO_SECRET_LOG"
  echo '❌ 无 JWT_SECRET 的生产镜像未因预期门禁及时拒绝启动' >&2
  exit 1
fi
rm -f "$NO_SECRET_LOG"

# 全新库也必须证明 8 个历史账号的两个公开旧口令全部被拒绝。
for username in admin cangguan jishuyuan1 jishuyuan2 yishi1 yishi2 caigou caiwu; do
  for old_password in admin123 'CoreOne2026!'; do
    docker compose exec -T -e CHECK_USERNAME="$username" -e CHECK_PASSWORD="$old_password" backend \
      node --input-type=module -e '
        const response = await fetch("http://127.0.0.1:3001/api/v1/auth/login", {
          method: "POST", headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ username: process.env.CHECK_USERNAME, password: process.env.CHECK_PASSWORD }),
        });
        process.exit(response.status === 401 ? 0 : 1);
      '
  done
done

# 在 backend 内部验证新凭据；口令只经 stdin 传入。
printf '%s' "$ADMIN_VERIFY_PASSWORD" | docker compose exec -T backend node --input-type=module -e '
  let password = ""; for await (const chunk of process.stdin) password += chunk;
  const response = await fetch("http://127.0.0.1:3001/api/v1/auth/login", {
    method: "POST", headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ username: "admin", password }),
  });
  process.exit(response.status === 200 ? 0 : 1);
'

# backend 安全验收通过后才启动 frontend，并通过 frontend 代理复验健康与关键登录路径。
docker compose up -d --no-deps --no-build frontend
test "$(docker inspect --format '{{.Image}}' coreone-frontend)" = "$FRONTEND_IMAGE_ID"
FRONTEND_PORT="$(docker compose port frontend 80 | sed -n '1s/.*://p')"
case "$FRONTEND_PORT" in
  ''|*[!0-9]*) echo '❌ 无法确定 frontend 宿主发布端口' >&2; exit 1 ;;
esac
test "$FRONTEND_PORT" = "$COREONE_PORT" \
  || { echo "❌ frontend 端口漂移：批准 ${COREONE_PORT}，实际 ${FRONTEND_PORT}" >&2; exit 1; }
FRONTEND_READY=0
for _ in $(seq 1 60); do
  if curl --max-time 5 -fsS "http://127.0.0.1:${FRONTEND_PORT}/health" >/dev/null; then
    FRONTEND_READY=1
    break
  fi
  FRONTEND_STATUS="$(docker inspect --format '{{.State.Status}}' coreone-frontend 2>/dev/null || true)"
  case "$FRONTEND_STATUS" in exited|dead) echo '❌ frontend 启动失败' >&2; exit 1 ;; esac
  sleep 2
done
test "$FRONTEND_READY" -eq 1 || { echo '❌ frontend 就绪超时' >&2; exit 1; }
docker compose exec -T backend node --input-type=module -e '
  const response = await fetch("http://frontend/health", { signal: AbortSignal.timeout(5000) });
  process.exit(response.ok ? 0 : 1);
'
printf '%s' "$ADMIN_VERIFY_PASSWORD" | docker run --rm -i --network host \
  -e CHECK_PORT="$FRONTEND_PORT" --entrypoint node "$BACKEND_IMAGE_ID" --input-type=module -e '
    let password = ""; for await (const chunk of process.stdin) password += chunk;
    const response = await fetch(`http://127.0.0.1:${process.env.CHECK_PORT}/api/v1/auth/login`, {
      method: "POST", headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ username: "admin", password }),
      signal: AbortSignal.timeout(5000),
    });
    process.exit(response.status === 200 ? 0 : 1);
  '

# 全新库没有可供轮换的历史 JWT，会话失效验收明确为 N/A；其余安全探针全部通过后才标记 release。
ln -sfn "$RELEASE_DIR" /opt/coreone/current
DEPLOYMENT_COMPLETE=1
trap - EXIT HUP INT TERM
unset ADMIN_VERIFY_PASSWORD JWT_SECRET
echo '✅ 全新部署通过：批准 SHA/镜像、backend/frontend 健康、无 secret 拒启、16 次旧口令拒绝和 admin 登录均已验证；旧 JWT 验收 N/A。'

# 查看状态；如需持续日志再单独执行 docker compose logs -f。
docker compose ps
```

若首次创建 volume 后在验收阶段失败，保留的 volume 属于“未批准候选数据”，不能直接重跑快速部署，
也不能假装它是带旧 token 的历史生产库。保持服务离线并留存诊断后，由变更负责人明确批准“销毁后
全新重建”或“按专用恢复步骤继续验收”；本手册不会自动删除任何 volume。

### 3. 访问系统

打开浏览器访问 `http://your-server-ip:<批准的 COREONE_PORT>`（未另行批准时默认 `8080`）。

登录：**系统不再内置默认账号/默认口令**（旧默认 `admin/admin123` 曾泄露，已移除）。全新生产
部署必须使用上一步受控输入的 `ADMIN_INITIAL_PASSWORD` 创建 admin；发布脚本会在切流前验证该
凭据，并强制重建 backend 清除容器配置中的初始口令。首次登录后按密码策略立即轮换；若 admin
创建或验证失败，部署不会被标记完成。`reset-passwords` **只更新已存在账号**，不会创建 admin。

## 旧生产库安全升级（必须先改密，再启动新服务）

新版本会在生产启动时有界检查以下 8 个历史默认账号中仍活跃、未删除的账号：
`admin`、`cangguan`、`jishuyuan1`、`jishuyuan2`、`yishi1`、`yishi2`、`caigou`、`caiwu`。八者中只要任一账号仍能
匹配已泄露口令 `admin123` 或 `CoreOne2026!`，backend 就会拒绝启动；这是有意的
fail-closed，不能通过反复重启绕过。

升级复用旧 `coreone-data` volume 时，严格按以下顺序执行。备份完成并通过 SQLite
`integrity_check` 前不得改密或启动新代码；任何 `docker compose down` 都**不要加 `-v`**。

```bash
set -euo pipefail
: "${APPROVED_SHA:?先设置本次获批的完整 40 位 SHA}"
RELEASE_DIR="/opt/coreone/releases/${APPROVED_SHA}"
test -d "$RELEASE_DIR" || { echo '❌ 批准 SHA 的 release 目录不存在' >&2; exit 1; }
cd "$RELEASE_DIR"
test "$(cat DEPLOYED_SHA)" = "$APPROVED_SHA" || { echo '❌ release 目录与批准 SHA 不一致' >&2; exit 1; }
export COMPOSE_PROJECT_NAME=coreone
export COREONE_IMAGE_TAG="$APPROVED_SHA"
export COREONE_ALLOW_DATABASE_CREATE=0
# 显式空值覆盖可能残留在 .env 里的初始口令；旧库升级绝不允许创建空库。
export ADMIN_INITIAL_PASSWORD=''

# 本节必须在同一个远端 shell 中从头执行到尾。开始停服后，任一步失败都让候选版本保持离线；
# 只保留生产 volume 与备份供排查，绝不自动回启已知存在风险的旧镜像。
DEPLOYMENT_MUTATED=0
DEPLOYMENT_COMPLETE=0
deployment_abort() {
  local status=$?
  local cleanup_ok=1
  trap - EXIT HUP INT TERM
  if [ "${DEPLOYMENT_MUTATED:-0}" -eq 1 ] && [ "${DEPLOYMENT_COMPLETE:-0}" -ne 1 ]; then
    echo '❌ 升级验收未完成；正在停止候选服务，生产 volume 与备份均保留。' >&2
    if ! docker compose down --remove-orphans; then
      echo '⚠️ compose down 失败，尝试按固定容器名强制停止。' >&2
      docker rm -f coreone-frontend coreone-backend >/dev/null 2>&1 || cleanup_ok=0
    fi
    if ! docker info >/dev/null 2>&1; then
      cleanup_ok=0
    else
      for container in coreone-frontend coreone-backend; do
        if docker container inspect "$container" >/dev/null 2>&1 \
          && [ "$(docker inspect --format '{{.State.Running}}' "$container")" = 'true' ]; then
          cleanup_ok=0
        fi
      done
    fi
    if [ "$cleanup_ok" -ne 1 ]; then
      echo '🚨 无法确认候选容器已停止；必须保持外部维护模式并人工处置，不能宣称回滚完成。' >&2
    fi
  fi
  unset RESET_ADMIN_PASSWORD RESET_CANGGUAN_PASSWORD RESET_JISHUYUAN1_PASSWORD
  unset RESET_JISHUYUAN2_PASSWORD RESET_YISHI1_PASSWORD RESET_YISHI2_PASSWORD
  unset RESET_CAIGOU_PASSWORD RESET_CAIWU_PASSWORD NEW_ADMIN_PASSWORD
  unset OLD_ACCESS_TOKEN OLD_REFRESH_TOKEN CURRENT_JWT_SECRET ROTATED_JWT_SECRET JWT_SECRET
  [ "$status" -ne 0 ] || status=1
  exit "$status"
}
trap deployment_abort EXIT HUP INT TERM

# 1. 先从正式密钥管理器读取“当前”JWT secret，仅用于 Compose 解析；不回显、不写命令历史。
read -rsp "Current JWT secret from the approved secret manager: " CURRENT_JWT_SECRET && echo
test -n "$CURRENT_JWT_SECRET" || { echo '❌ 未读取到当前 JWT secret' >&2; exit 1; }
export JWT_SECRET="$CURRENT_JWT_SECRET"

# 从仍在运行的旧 frontend 捕获真实宿主端口，并明确确认它仍是本次批准的负载均衡目标。
COREONE_PORT="$(docker compose port frontend 80 | sed -n '1s/.*://p')"
case "$COREONE_PORT" in
  ''|*[!0-9]*) echo '❌ 无法捕获现生产 frontend 宿主端口' >&2; exit 1 ;;
esac
[ "$COREONE_PORT" -ge 1 ] && [ "$COREONE_PORT" -le 65535 ] || exit 1
read -rp "Confirm approved load-balancer target port [type ${COREONE_PORT}]: " APPROVED_FRONTEND_PORT
test "$APPROVED_FRONTEND_PORT" = "$COREONE_PORT" \
  || { echo '❌ 负载均衡目标端口未获确认，停止升级。' >&2; exit 1; }
unset APPROVED_FRONTEND_PORT
export COREONE_PORT

# 从部署前仍有效的受控会话取得一对旧 access/refresh token；二者都不回显、不打印。
read -rsp "Pre-rotation access token for invalidation check: " OLD_ACCESS_TOKEN && echo
read -rsp "Pre-rotation refresh token for invalidation check: " OLD_REFRESH_TOKEN && echo
test -n "$OLD_ACCESS_TOKEN" && test -n "$OLD_REFRESH_TOKEN" \
  || { echo '❌ 缺少旧 access/refresh token 验收样本' >&2; exit 1; }

# 必须在旧服务仍运行时先证明两个样本都有效，避免用过期/伪造 token 制造轮换假绿。
printf '%s' "$OLD_ACCESS_TOKEN" | docker compose exec -T backend node --input-type=module -e '
  let token = ""; for await (const chunk of process.stdin) token += chunk;
  const response = await fetch("http://127.0.0.1:3001/api/v1/auth/me/capabilities", { headers: { Authorization: `Bearer ${token}` } });
  process.exit(response.status === 200 ? 0 : 1);
'
printf '%s' "$OLD_REFRESH_TOKEN" | docker compose exec -T backend node --input-type=module -e '
  let token = ""; for await (const chunk of process.stdin) token += chunk;
  const response = await fetch("http://127.0.0.1:3001/api/v1/auth/refresh", {
    method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ refreshToken: token }),
  });
  process.exit(response.status === 200 ? 0 : 1);
'

# 现在才从正式密钥管理器读取已批准的新 secret；必须与当前值不同。本任务不在此生成或展示密钥。
read -rsp "Rotated JWT secret from the approved secret manager: " ROTATED_JWT_SECRET && echo
test -n "$ROTATED_JWT_SECRET" && test "$ROTATED_JWT_SECRET" != "$CURRENT_JWT_SECRET" \
  || { echo '❌ 新 JWT secret 缺失或未发生轮换' >&2; exit 1; }
export JWT_SECRET="$ROTATED_JWT_SECRET"

### 失败后离线 roll-forward（生产执行前置条件）

上面的失败 trap 会安全地保留 production volume/备份并让候选服务维持离线，但它不会回启
已知存在风险的旧镜像，也不会猜测目标环境的负载均衡、备份库或密钥管理器状态。因此，**合并
代码不等于已获生产部署授权**：执行旧库升级前，变更负责人必须在目标环境的受控变更系统中准备
并演练一份可执行的离线 roll-forward 变更单；没有该变更单时禁止开始停服步骤。

变更单至少要固定并复核以下检查点：获批 40 位 SHA 与前后端镜像 ID、真实 production volume、
已通过 `integrity_check` 与 SHA-256 的备份、已批准宿主端口、密钥管理器中的已轮换 JWT secret、
轮换前 access/refresh 均为 `200` 的证据和同一对受控 token 样本。续跑只能使用获批安全镜像，
再次执行 8 账号原子改密；随后按本节相同顺序完成 backend-only 健康、无 secret 拒启、16 次旧
口令 `401`、旧 access/refresh `401`、新 admin 登录 `200`、frontend 宿主健康与代理登录，全部
通过后才切换 `current` 和恢复流量。任何证据缺失或探针失败都继续保持维护模式，并升级审批；
不得用回启 PR #119 之前的镜像作为“临时恢复”。变更单不得记录真实 secret、token 或口令正文。

### 修改 JWT Secret（生产环境必须改！）

生产环境**禁止**只执行一条 `docker compose up --force-recreate backend` 就宣布轮换完成；本仓库
不提供这种会产生“假失效”结论的捷径。每次 JWT 轮换都必须在同一受控 shell 中复用上文
“旧生产库安全升级”的完整门禁：先用当前 secret 证明旧 access 与 refresh 均为 `200`；从正式
密钥管理器读取批准的新值并证明新旧不同；保持 `COREONE_ALLOW_DATABASE_CREATE=0` 和空初始口令；
摘除流量并重建固定 SHA 的 backend；等待 `healthy`；证明旧 access 与 refresh 均精确为 `401`；
最后经 frontend 宿主发布端口证明健康检查和新凭据登录为 `200`。任一探针失败时保持 backend
离线并维持维护模式。若无法取得轮换前的有效 token 对或批准的新 secret，停止变更并升级审批，
不得用过期 token、空 secret 或服务超时替代验收。

### 重置已存在账号口令

历史 8 账号必须按上文“旧生产库安全升级”一次性原子轮换，不能只改 admin 后就宣布止血。
脚本仍支持 `RESET_ADMIN_PASSWORD` 和高级批量变量 `RESET_PASSWORDS_JSON`，但生产事故处置
优先使用上文 8 个独立变量：输入不回显、不会把实际口令写进 shell 历史，也不需要手写 JSON。
脚本只更新已存在账号，不会创建账号；所有目标在一个事务中处理，任一失败则整体回滚。

docker compose run --rm --no-deps -v "$PWD/backup:/backup:ro" -e BACKUP_NAME="$BACKUP_NAME" \
  --entrypoint sh backend -c \
  'node --experimental-sqlite --input-type=module -e '\''import { DatabaseSync } from "node:sqlite"; const db = new DatabaseSync(`/backup/${process.env.BACKUP_NAME}`, { readOnly: true }); const row = db.prepare("PRAGMA integrity_check").get(); db.close(); if (row.integrity_check !== "ok") throw new Error("backup integrity_check failed")'\'''
sha256sum "backup/$BACKUP_NAME" | tee "backup/$BACKUP_NAME.sha256"

# 备份完成后恢复同一批准 SHA 的服务并验健康；到此为“仅备份”流程终点，不会继续执行恢复。
BACKEND_IMAGE_ID="$(docker image inspect --format '{{.Id}}' "coreone-backend:${APPROVED_SHA}")"
docker compose up -d --no-deps --no-build backend
for _ in $(seq 1 60); do
  BACKEND_STATUS="$(docker inspect --format '{{if .State.Health}}{{.State.Health.Status}}{{else}}{{.State.Status}}{{end}}' coreone-backend 2>/dev/null || true)"
  [ "$BACKEND_STATUS" = 'healthy' ] && break
  case "$BACKEND_STATUS" in exited|dead) echo '❌ backend 重启失败；保持离线并排查' >&2; exit 1 ;; esac
  sleep 2
done
test "${BACKEND_STATUS:-}" = 'healthy'
test "$(docker inspect --format '{{.Image}}' coreone-backend)" = "$BACKEND_IMAGE_ID"
unset JWT_SECRET
echo "✅ 备份完成并已恢复 backend：backup/$BACKUP_NAME"
```

恢复是独立的高风险操作，必须另开变更窗口。先把 `RESTORE_NAME` 改成真实、已审批的文件名；
下面流程成功后也会保持业务离线，直到重新完成 8 账号原子改密和全部发布验收：

```bash
set -euo pipefail
cd /opt/coreone/current
APPROVED_SHA="$(cat DEPLOYED_SHA)"
[[ "$APPROVED_SHA" =~ ^[0-9a-f]{40}$ ]] || { echo '❌ DEPLOYED_SHA 非法' >&2; exit 1; }
export COMPOSE_PROJECT_NAME=coreone COREONE_IMAGE_TAG="$APPROVED_SHA"
export COREONE_ALLOW_DATABASE_CREATE=0 ADMIN_INITIAL_PASSWORD=''
read -rsp "Current JWT secret from the approved secret manager: " JWT_SECRET && echo
test -n "$JWT_SECRET" || exit 1
export JWT_SECRET

RESTORE_NAME='<verified-backup-file.db>'
test "$RESTORE_NAME" != '<verified-backup-file.db>' \
  || { echo '❌ 必须先填写真实的已审批备份文件名' >&2; exit 1; }
case "$RESTORE_NAME" in *[!A-Za-z0-9._-]*|'') echo '❌ 备份文件名非法' >&2; exit 1 ;; esac
sha256sum -c "backup/$RESTORE_NAME.sha256"
read -rp "Confirm all production traffic is drained and maintenance mode is active [type DRAINED]: " TRAFFIC_DRAIN_CONFIRM
test "$TRAFFIC_DRAIN_CONFIRM" = 'DRAINED' || { echo '❌ 未确认流量隔离' >&2; exit 1; }
docker compose stop frontend backend
docker compose down                         # 不得加 -v
COREONE_ALLOW_DATABASE_CREATE=0 docker compose run --rm --no-deps -v "$PWD/backup:/backup:ro" -e RESTORE_NAME="$RESTORE_NAME" \
  --entrypoint sh backend -c \
  'set -eu; test -s "/backup/$RESTORE_NAME"; cp "/backup/$RESTORE_NAME" /app/data/coreone.db; rm -f /app/data/coreone.db-wal /app/data/coreone.db-shm; chmod 600 /app/data/coreone.db'
docker compose run --rm --no-deps --entrypoint sh backend -c \
  'node --experimental-sqlite --input-type=module -e '\''import { DatabaseSync } from "node:sqlite"; const db = new DatabaseSync("/app/data/coreone.db", { readOnly: true }); const row = db.prepare("PRAGMA integrity_check").get(); db.close(); if (row.integrity_check !== "ok") throw new Error("restored database integrity_check failed")'\'''
unset JWT_SECRET
echo '✅ 快照已恢复且完整性通过；服务仍保持离线。现在执行 8 账号原子改密与上文完整发布验收。'
```

恢复的是发布前快照，可能重新带回已泄露的默认口令。因此恢复后必须保持业务服务下线，使用
当前获批安全镜像重新执行上文 **8 账号原子重置**，并再次完成全部验收。禁止启动 PR #119
之前的漏洞版本或旧镜像；若当前安全版本无法启动，应恢复数据、重新改密后保持下线，修复并
审批新的安全 SHA 后 roll-forward，而不是把漏洞代码作为回滚路径。

## 故障排查

| 现象 | 排查方法 |
|------|----------|
| 页面空白 404 | `docker compose logs frontend` 检查 Nginx 配置 |
| API 请求失败 | `docker compose logs backend` 检查后端是否启动 |
| 数据库权限错误 | `docker exec coreone-backend ls -la /app/data` |
| 端口被占用 | `netstat -tlnp \| grep 8080` 修改 `COREONE_PORT` |
| 容器无法启动 | `docker compose up`（不加 -d）查看实时错误 |

## 安全建议（生产环境）

1. **必须修改 JWT_SECRET** — 不要用默认值
2. **配置防火墙** — 只开放必要的端口
3. **启用 HTTPS** — 使用 Nginx + Let's Encrypt 证书
4. **定期备份数据库** — 建议每天自动备份
5. **限制 CORS** — 当前 `app.ts` 中 `cors()` 是开放配置，生产环境应限制为前端域名
